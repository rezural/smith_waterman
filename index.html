Smith–Waterman algorithm

    use smith_waterman;
    let mut smitty = smith_waterman::SmithWaterman::new( 
                                         "ACACACTA".to_string(),
                                         "AGCACACA".to_string());
    let alignment = smitty.align();
    println!("{:?}", smitty);
    println!("{:?}", alignment);
